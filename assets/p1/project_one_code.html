import matplotlib.pyplot as plt
genre_avg = mxmh_no_nulls.groupby('Fav genre')[['Anxiety', 'Depression', 'Insomnia', 'OCD']].mean()

bar_width = 0.2
x = range(len(genre_avg.index))

plt.figure(figsize=(16, 6)) 

# evenly space out the bars
plt.bar([i - 1.5 * bar_width for i in x], genre_avg["Anxiety"], width=bar_width, label="Anxiety", color="skyblue")
plt.bar([i - 0.5 * bar_width for i in x], genre_avg["Depression"], width=bar_width, label="Depression", color="lightcoral")
plt.bar([i + 0.5 * bar_width for i in x], genre_avg["Insomnia"], width=bar_width, label="Insomnia", color="lightseagreen")
plt.bar([i + 1.5 * bar_width for i in x], genre_avg["OCD"], width=bar_width, label="OCD", color="lightpink")

plt.xlabel("Favorite Genre")
plt.ylabel("Average Values")
plt.title("Average Values of Mental Health Parameters by Favorite Genre")

plt.legend()

plt.xticks(x, genre_avg.index)

plt.xticks(rotation=45)

plt.savefig("average_values_by_favorite_genre.png", dpi=300, bbox_inches="tight")

plt.tight_layout()
plt.show()
